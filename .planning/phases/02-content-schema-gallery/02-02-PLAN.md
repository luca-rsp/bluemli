```javascript
// scripts/prebuild-images.mjs
// Phase 2 (CNT-11, D-01..D-04): converts every src/content/gallery/<slug>/hero.*
// (HEIC, JPG, JPEG, PNG) into responsive WebP variants at public/gallery/<slug>/
// hero-{400,800,1600}.webp.
//
// Why pure-JS heic-convert (no libheif-dev apt step):
//   sharp's prebuilt binaries do NOT include HEIC support — they bundle a libvips
//   compiled WITHOUT libheif, regardless of what's on the system. Installing
//   libheif-dev only helps if you also rebuild sharp from source. heic-convert
//   is pure JS (heic-decode + jpeg-js + pngjs) so it works on any Node 22+
//   environment with zero system deps. See 02-RESEARCH.md §2 for the full
//   investigation.
//
// Why an explicit CI step (not an npm `prebuild` lifecycle hook):
//   pnpm 10 disables pre/post user-script hooks by default. A `prebuild` script
//   would silently skip unless `.npmrc` declares `enable-pre-post-scripts=true`.
//   The explicit script `prebuild:images` invoked from .github/workflows/ci.yml
//   is unambiguous. See 02-RESEARCH.md §3.
//
// Usage:
//   pnpm run prebuild:images        (local — produces gitignored variants)
//   In CI: invoked between Typecheck and Build steps.

import { readdir, readFile, mkdir } from 'node:fs/promises';
import { existsSync } from 'node:fs';
import { join, extname } from 'node:path';
import sharp from 'sharp';
import convert from 'heic-convert';

const GALLERY_SRC = './src/content/gallery';
const GALLERY_OUT = './public/gallery';
const WIDTHS = [400, 800, 1600];     // D-03: phone thumbnail / phone-detail + desktop-grid / Retina detail
const WEBP_QUALITY = 80;             // 02-RESEARCH.md §2: chosen for jewelry photo richness; tunable up to 80 per discretion

if (!existsSync(GALLERY_SRC)) {
  console.error(`FAIL: ${GALLERY_SRC} not found. Phase 2 expects per-slug folders here.`);
  process.exit(1);
}

const slugDirs = await readdir(GALLERY_SRC, { withFileTypes: true });

let processed = 0;
let skipped = 0;

for (const dirent of slugDirs) {
  if (!dirent.isDirectory()) continue;
  const slug = dirent.name;

  // Find hero.* — HEIC first (founder default from iPhone), then jpg/jpeg/png fallbacks.
  const heroExts = ['hero.heic', 'hero.jpg', 'hero.jpeg', 'hero.png'];
  let heroPath = null;
  for (const name of heroExts) {
    const candidate = join(GALLERY_SRC, slug, name);
    if (existsSync(candidate)) { heroPath = candidate; break; }
  }
  if (!heroPath) {
    console.warn(`[prebuild] WARNING: no hero.* found for ${slug}, skipping`);
    skipped++;
    continue;
  }

  const outDir = join(GALLERY_OUT, slug);
  await mkdir(outDir, { recursive: true });

  let sourceBuffer = await readFile(heroPath);

  // HEIC → JPEG buffer via pure-JS heic-convert (D-04 resolution per 02-RESEARCH.md §2).
  // CRITICAL: heic-convert returns ArrayBuffer; must wrap with Buffer.from() before
  // passing to sharp(). 02-PATTERNS.md "heic-convert buffer API" calls this out.
  if (extname(heroPath).toLowerCase() === '.heic') {
    const arrayBuffer = await convert({
      buffer: sourceBuffer,
      format: 'JPEG',
      quality: 1,                    // max-quality intermediate; sharp re-encodes to WebP
    });
    sourceBuffer = Buffer.from(arrayBuffer);
  }

  // Generate 3 WebP widths. fit: 'inside' preserves aspect ratio (D-03 — no crop in
  // the pipeline; cropping is a render-time CSS concern via aspect-ratio + object-fit).
  // withoutEnlargement: true: if source is smaller than target, output stays at source size.
  for (const w of WIDTHS) {
    const outPath = join(outDir, `hero-${w}.webp`);
    await sharp(sourceBuffer)
      .resize(w, null, { fit: 'inside', withoutEnlargement: true })
      .webp({ quality: WEBP_QUALITY })
      .toFile(outPath);
    console.log(`[prebuild] ${slug}/hero-${w}.webp`);
  }
  processed++;
}

console.log(`[prebuild] Done. ${processed} piece(s) processed, ${skipped} skipped.`);
```

**Determinism notes:**
- Running this script twice MUST produce byte-identical output (sharp's WebP encoder is deterministic at fixed quality/options).
- The script is destructive at the per-file level (overwrites existing `hero-*.webp`), but does NOT delete `public/gallery/<slug>/` directories that no longer have a source — that's intentional (founder removing a slug means the variants persist until a clean build wipes `public/gallery/` via the gitignore rule + fresh `git clone`).
- DO NOT add parallelism (`Promise.all`) — sharp is single-threaded per call and 6 pieces × 3 widths = 18 sequential transforms complete in <30s on ubuntu-latest. Adding parallelism complicates error handling for marginal speedup.
- DO NOT add LQIP/blurhash placeholders (deferred per CONTEXT.md `<deferred>` — UI-SPEC says the `--cream-200` background fallback is sufficient).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Founder iPhone HEIC → CI runner | Founder uploads HEIC files via GitHub web UI; the CI runner reads and decodes them |
| heic-convert (pure JS) → sharp (native binding) → filesystem | Decoded JPEG buffer flows to sharp's native libvips for WebP encoding |
| public/gallery/ build artifacts → Cloudflare Static Assets | WebP variants ship as public, uncapped static assets |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04 | Denial of Service | Malformed HEIC parse-bombs (deeply nested HEIF boxes, bombs that expand to absurd dimensions) | mitigate | heic-convert is the JS surface; if it throws on a malformed input, the script exits non-zero and the CI build fails (the failing slug name is in the script's preceding `console.log`). Add no try/catch around the conversion — surface the error loudly so the failing slug is obvious in the CI log. The CI job has a `timeout-minutes: 10` cap (existing) so a hung decode is bounded. |
| T-02-05 | Information Disclosure | EXIF metadata (GPS coordinates, device serial, timestamps) leaking from HEIC into served WebP | mitigate | sharp's WebP encoder strips EXIF/IPTC/XMP metadata by default unless `withMetadata()` is called — the script does NOT call `withMetadata()`, so all metadata is stripped. Verify in acceptance criteria via `exiftool` or `webpinfo` if available; otherwise rely on sharp's documented default behavior. |
| T-02-06 | Tampering | A contributor commits a binary file disguised as `hero.heic` (e.g., a shell script with .heic extension) | accept | The script reads the file as bytes and passes them to heic-convert; if not valid HEIC, heic-convert throws and the build fails. No code execution occurs. The HEIC source files live in the repo and are reviewed in PRs like any other file. |
</threat_model>
