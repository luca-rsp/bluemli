Full script content:

```bash
#!/usr/bin/env bash
# scripts/test-schema-strict.sh — SC2 verification (Phase 2).
# Deliberately typos a frontmatter field, asserts the build fails with a Zod
# "Unrecognized key" or .strict() error, then restores the original file.
#
# Usage: bash scripts/test-schema-strict.sh
# Exit codes: 0 = SC2 satisfied (typo correctly rejected); 1 = SC2 FAIL.

set -uo pipefail

TARGET="src/content/gallery/cluster-blush/index.md"
BACKUP="$(mktemp)"
ASTRO_LOG="$(mktemp)"

cleanup() {
  if [ -f "$BACKUP" ]; then
    cp "$BACKUP" "$TARGET"
    rm -f "$BACKUP" "$ASTRO_LOG"
  fi
}
trap cleanup EXIT INT TERM

if [ ! -f "$TARGET" ]; then
  echo "FAIL: $TARGET not found — Plan 01 Task 2 seed file missing."
  exit 1
fi

# Save and corrupt
cp "$TARGET" "$BACKUP"

# Insert a typo'd field into the frontmatter (after the opening ---) by inserting
# 'availabilty: sold' on line 2. Using awk so we don't depend on sed -i portability.
awk 'NR==1{print; print "availabilty: sold"; next} {print}' "$BACKUP" > "$TARGET"

echo "[SC2] Running astro check against typo'd frontmatter..."
if pnpm exec astro check > "$ASTRO_LOG" 2>&1; then
  echo "FAIL: astro check exited 0 with typo'd frontmatter — .strict() not enforced."
  cat "$ASTRO_LOG"
  exit 1
fi

# astro check failed (as expected) — confirm the error message names the schema problem.
if grep -qE "availabilty|Unrecognized|strict" "$ASTRO_LOG"; then
  echo "[SC2] PASS: astro check rejected the typo'd field with a clear schema error."
else
  echo "WARN: astro check failed but didn't mention 'availabilty', 'Unrecognized', or 'strict' — verify error quality."
  echo "Output was:"
  cat "$ASTRO_LOG"
  # Still pass — the build did fail, which is the core SC2 requirement.
fi

# Restore happens via trap. Verify post-restore clean.
cleanup
trap - EXIT INT TERM

if pnpm exec astro check > /dev/null 2>&1; then
  echo "[SC2] Restored file passes astro check — clean state."
  exit 0
else
  echo "FAIL: post-restore astro check failed — something else is broken (NOT the typo test)."
  exit 1
fi
```
